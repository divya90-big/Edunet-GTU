d = p.read_csv('green_skills_dataset.csv')

print(d.head())
print('The last 5 rows of the dataset are:')
print(d.tail())
print('The dataset information is:')
d.info()
print('The number of null values in each column is:')
print(d.isnull().sum())
if d.isnull().sum().any():
    d = d.ffill()
    print('After filling the null values, the dataset is:')
    print(d)
